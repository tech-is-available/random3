const express = require("express");
const app = express();

app.get("/add/:num1/:num2/:num3", (req, res) =>
{
  const num1 = req.params.num1;
  const num2 = req.params.num2;
  const num3 = req.params.num3;
  const result = Number(num1) + Number(num2) + Number(num3);
  res.send( "ANS=" + result);
});


app.listen(3000, () =>
console.log("http://localhost:3000/add/3/5/8")
);
