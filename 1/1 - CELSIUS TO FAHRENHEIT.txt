const express = require("express");
const app = express();

app.get("/convert", (req, res) => 
{
  const fahrenheit = Number(req.query.f);
  const celsius = (fahrenheit - 32) * 5 / 9;
  res.send("converted string is : " + celsius);
});


app.listen(3000, () =>
console.log("http://localhost:3000/convert?f=98.6")
);
