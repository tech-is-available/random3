const express = require("express");
const app = express();

app.get("/convert", (req, res) => 
{
  const num = Number(req.query.num);

  let fact = 1;
  for (let i = 1; i <= num; i++) {
    fact = fact * i;
  }
  res.send("Factorial is : " + fact);
});


app.listen(3000, () =>
console.log("http://localhost:3000/convert?num=5")
);
